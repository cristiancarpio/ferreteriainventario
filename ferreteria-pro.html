// ============================================ // FERRETERÍA PRO — Google Apps Script // Pegá este código en tu Apps Script // ============================================ function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(data.sheet); if (!sheet) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:"Hoja no encontrada: "+data.sheet})) .setMimeType(ContentService.MimeType.JSON); } var row = data.rowIndex; // Columnas: A=Nombre B=Código C=Proveedor D=Mayorista E=Venta F=Stock G=Categoría if (data.field === "mayorista") { sheet.getRange(row, 4).setValue(data.value); } else if (data.field === "venta") { sheet.getRange(row, 5).setValue(data.value); } else if (data.field === "stock") { sheet.getRange(row, 6).setValue(data.value); } else if (data.field === "all") { sheet.getRange(row, 4).setValue(data.mayorista); sheet.getRange(row, 5).setValue(data.venta); sheet.getRange(row, 6).setValue(data.stock); } return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:"Ferretería Pro API activa"})) .setMimeType(ContentService.MimeType.JSON); }
